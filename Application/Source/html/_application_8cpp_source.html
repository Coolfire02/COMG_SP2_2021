84 bool Application::GetMouseUpdate()
85 {
86  if (glfwGetInputMode(m_window, GLFW_CURSOR) != GLFW_CURSOR_DISABLED)
87  return false;
88 
89  glfwGetCursorPos(m_window, &mouse_current_x, &mouse_current_y);
90 
91  // Calculate the difference in positions
92  mouse_diff_x = mouse_current_x - mouse_last_x;
93  mouse_diff_y = mouse_current_y - mouse_last_y;
94 
95  //Calculate the yaw and pitch
96  camera_yaw = (float)mouse_diff_x * 0.0174555555555556f;// * 3.142f / 180.0f;
97  camera_pitch = mouse_diff_y * 0.0174555555555556f;// 3.142f / 180.0f );
98 
99  // Do a wraparound if the mouse cursor has gone out of the deadzone
100  if ((mouse_current_x < m_window_deadzone) || (mouse_current_x > m_window_width - m_window_deadzone))
101  {
102  mouse_current_x = m_window_width >> 1;
103  glfwSetCursorPos(m_window, mouse_current_x, mouse_current_y);
104  }
105  if ((mouse_current_y < m_window_deadzone) || (mouse_current_y > m_window_height - m_window_deadzone))
106  {
107  mouse_current_y = m_window_height >> 1;
108  glfwSetCursorPos(m_window, mouse_current_x, mouse_current_y);
109  }
110 
111  // Store the current position as the last position
112  mouse_last_x = mouse_current_x;
113  mouse_last_y = mouse_current_y;
114 
115  return false;
116 }

223 void UIManager::Update(Scene* scene, double dt)
224 {
225  bManagers[currentMenu]->Update(scene, dt);
226  activeMenus[UI_E_TO_INTERACT] = false;
227  if (Game::settings[SETTING_FPS]) {
228  std::ostringstream ss;
229  ss.precision(3);
230  ss << Game::FPS;
231  Button* bt = this->getByTypeBM(UI_FPS)->getButtonByName("FPS");
232  bt->setText(ss.str());
233  if (Game::FPS < 20) {
234  bt->getTextObject()->setColor(Color(0.85f, 0.1f, 0.1f));
235  }
236  else if (Game::FPS < 40) {
237  bt->getTextObject()->setColor(Color(0.85f, 0.85f, 0.1f));
238  }
239  else {
240  bt->getTextObject()->setColor(Color(0.1f, 0.9f, 0.1f));
241  }
242  activeMenus[UI_FPS] = true;
243  }
244  else {
245  activeMenus[UI_FPS] = false;
246  }
247  if (uiActive == true)
248  {
249  if (Application::IsKeyPressed('L') && (currentMenu != UI_ITEM_INVENTORY && currentMenu != UI_INTERACTION && currentMenu != UI_MAIN_MENU) && !(currentMenu == UI_ITEM_INVENTORY || currentMenu == UI_WEAPON_INVENTORY || currentMenu == UI_GARAGE_INVENTORY))
250  {
251  if (nextKeyPress < Game::gElapsedTime)
252  { //Means can press
253  setCurrentUI(UI_ITEM_INVENTORY);
254  nextKeyPress = Game::gElapsedTime + 1.0f;
255  }
256  }
257  else if (Application::IsKeyPressed('L') && (currentMenu == UI_ITEM_INVENTORY || currentMenu == UI_WEAPON_INVENTORY || currentMenu == UI_GARAGE_INVENTORY))
258  {
259  if (nextKeyPress < Game::gElapsedTime)

262  nextKeyPress = Game::gElapsedTime + 1.0f;

266  switch (currentMenu) {

272  setCurrentUI(UI_WEAPON_INVENTORY);
273  }

275  {
276  setCurrentUI(UI_GARAGE_INVENTORY);
277  }
278  for (int i = 0; i < ITEM_AMOUNT; i++)
279  {
280  if (buttonCollide->buttonClicked->getName() == ("UIItemInventorySlot" +std::to_string(i + 1)) && buttonCollide->justHovered)
281  {
282  Game::inv.getItemInventory()->setCurrentItem(i);
283  }
284  }
285  break;
286  case UI_WEAPON_INVENTORY:
287  if (buttonCollide->buttonClicked->getName() == "UIItemsInventory" && buttonCollide->justClicked) //Items button
288  {
289  setCurrentUI(UI_ITEM_INVENTORY);
290  }
291  if (buttonCollide->buttonClicked->getName() == "UIGarageInventory" && buttonCollide->justClicked) //Garage Button
292  {
293  setCurrentUI(UI_GARAGE_INVENTORY);
294  }
295  if (Game::inv.getWeaponVector().size() > 0)
296  {
297  for (int i = 0; i < Game::inv.getWeaponVector().size(); i++)
298  {

306  case UI_GARAGE_INVENTORY:

313  setCurrentUI(UI_WEAPON_INVENTORY);

317  for (int i = 0; i < Game::inv.getGarageVector().size(); i++)

342  setCurrentUI(UI_SETTINGS);

344  break;

349  buttonCollide->buttonClicked->setQuadImage((GEOMETRY_TYPE)((int)SETTINGS_BUTTON_OFF + Game::settings[SETTINGS::SETTING_SAND_BOX_MODE]));

354  buttonCollide->buttonClicked->setQuadImage((GEOMETRY_TYPE)((int)SETTINGS_BUTTON_OFF + Game::settings[SETTINGS::SETTING_FPS]));
355  }
356  break;

359  {
360  setCurrentUI(UI_GENERAL);
361  }
362  if (buttonCollide->buttonClicked->getName() == "PauseMenuQuitButton" && buttonCollide->justClicked)//Main Menu quit button
363  {
364  setCurrentUI(UI_MAIN_MENU);
365  Game::setPrevSceneENUM(Game::activeScene);
366  Game::switchScene(S_UI);

368  else if (buttonCollide->buttonClicked->getName() == "SettingsButton" && buttonCollide->justClicked)
369  {
370  setCurrentUI(UI_SETTINGS);
371  }
372  break;
373  case UI_INTERACTION:
374  if (Game::iManager.getQueue().Top()->interactionChoices.empty()) {
375  if (buttonCollide->buttonClicked->getName() == "InteractionButton" && buttonCollide->justClicked) {
376  Game::iManager.nextInteraction(Game::iManager.getQueue().Top()->nextInteractionKey);

381  std::stringstream ss;
382  ss << "Choice" << i + 1;
383  if (buttonCollide->buttonClicked->getName() == ss.str() && buttonCollide->justClicked) {
384  for (auto& entry : Game::iManager.getQueue().Top()->interactionChoices[i]->postInteractionCMD) {
385  Game::iManager.runCommand(*entry);
386  }
387  Game::iManager.nextInteraction(Game::iManager.getQueue().Top()->interactionChoices[i]->nextInteractionKey);
388  break;
389  }
390  }
391  }
392  break;
393  }
394  }
395  if (Game::uiManager.getCurrentMenu() == UI_GENERAL)
396  {
397 
398  if (Game::inv.getActiveWeapon() == nullptr) {
399  Game::uiManager.getByTypeBM(UI_GENERAL)->getButtonByName("AmmoCount")->disable();
400  Game::uiManager.getByTypeBM(UI_GENERAL)->getButtonByName("TotalAmmoCount")->disable();

410  Game::uiManager.getByTypeBM(UI_GENERAL)->getButtonByName("Cash")->setText("$" + std::to_string(Game::cash));
411  }
412  }
